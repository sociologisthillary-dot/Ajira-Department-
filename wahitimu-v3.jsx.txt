function ProgressBar({ step }) {
  const steps = [
    { key: "upload", label: "Pakia", icon: "📤" },
    { key: "processing", label: "Inafanya", icon: "⚙️" },
    { key: "preview", label: "Hakiki", icon: "👁️" },
    { key: "payment", label: "Lipa", icon: "💳" },
    { key: "download", label: "Pakua", icon: "⬇️" },
  ];
  const order = steps.map((s) => s.key);
  const curr = order.indexOf(step);

  return (
    <div style={{ display: "flex", alignItems: "center", justifyContent: "center", gap: 0, marginBottom: 32, flexWrap: "wrap", rowGap: 8 }}>
      {steps.map((s, i) => {
        const done = curr > i;
        const active = curr === i;
        return (
          <div key={s.key} style={{ display: "flex", alignItems: "center" }}>
            <div style={{
              display: "flex", flexDirection: "column", alignItems: "center", gap: 4,
            }}>
              <div style={{
                width: 38, height: 38, borderRadius: "50%",
                background: done ? "#1e3a8a" : active ? "linear-gradient(135deg,#1e3a8a,#2563eb)" : "#e2e8f0",
                display: "flex", alignItems: "center", justifyContent: "center",
                fontSize: done ? 16 : 15,
                boxShadow: active ? "0 0 0 4px rgba(30,58,138,0.2)" : "none",
                transition: "all .3s",
              }}>
                {done ? "✓" : <span style={{ fontSize: 15 }}>{s.icon}</span>}
              </div>
              <span style={{ fontSize: 10, color: active ? "#1e3a8a" : done ? "#1e3a8a" : "#94a3b8", fontWeight: active ? 700 : 500 }}>{s.label}</span>
            </div>
            {i < steps.length - 1 && (
              <div style={{ width: 28, height: 2, background: done ? "#1e3a8a" : "#e2e8f0", marginBottom: 18, transition: "all .3s" }} />
            )}
          </div>
        );
      })}
    </div>
  );
}

function SignaturePad({ onSave, savedSignature }) {
  const canvasRef = useRef(null);
  const [isDrawing, setIsDrawing] = useState(false);
  const [hasDrawn, setHasDrawn] = useState(!!savedSignature);
  const [mode, setMode] = useState("draw"); // "draw" or "type"
  const [typedName, setTypedName] = useState("");
  const [font, setFont] = useState("'Brush Script MT', cursive");

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    ctx.lineWidth = 2.2;
    ctx.lineCap = "round";
    ctx.strokeStyle = "#1e3a8a";
    if (savedSignature && mode === "draw") {
      const img = new Image();
      img.onload = () => ctx.drawImage(img, 0, 0);
      img.src = savedSignature;
    }
  }, []);

  function getPos(e, canvas) {
    const rect = canvas.getBoundingClientRect();
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    return { x: clientX - rect.left, y: clientY - rect.top };
  }

  function start(e) {
    e.preventDefault();
    setIsDrawing(true);
    setHasDrawn(true);
    const canvas = canvasRef.current;
    const ctx = canvas.getContext("2d");
    const { x, y } = getPos(e, canvas);
    ctx.beginPath();
    ctx.moveTo(x, y);
  }

  function draw(e) {
    if (!isDrawing) return;
    e.preventDefault();
    const canvas = canvasRef.current;
    const ctx = canvas.getContext("2d");
    const { x, y } = getPos(e, canvas);
    ctx.lineTo(x, y);
    ctx.stroke();
  }

  function stop() {
    if (!isDrawing) return;
    setIsDrawing(false);
    saveCanvas();
  }

  function clearCanvas() {
    const canvas = canvasRef.current;
    const ctx = canvas.getContext("2d");
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    setHasDrawn(false);
    onSave(null);
  }

  function saveCanvas() {
    const canvas = canvasRef.current;
    onSave(canvas.toDataURL("image/png"));
  }

  function saveTyped(name, fontFamily) {
    if (!name.trim()) { onSave(null); return; }
    const canvas = document.createElement("canvas");
    canvas.width = 500; canvas.height = 150;
    const ctx = canvas.getContext("2d");
    ctx.fillStyle = "#1e3a8a";
    ctx.font = `48px ${fontFamily}`;
    ctx.textBaseline = "middle";
    ctx.fillText(name, 20, 75);
    onSave(canvas.toDataURL("image/png"));
  }

  const fonts = [
    { id: "'Brush Script MT', cursive", label: "Mtindo 1" },
    { id: "'Lucida Handwriting', cursive", label: "Mtindo 2" },
    { id: "'Segoe Script', cursive", label: "Mtindo 3" },
    { id: "Georgia, serif", label: "Rasmi" },
  ];

  return (
    <div>
      {/* Mode toggle */}
      <div style={{ display: "flex", gap: 8, marginBottom: 12 }}>
        <button onClick={() => setMode("draw")} style={{
          flex: 1, padding: "8px 12px", borderRadius: 8, cursor: "pointer", fontSize: 12, fontWeight: 700,
          border: `2px solid ${mode === "draw" ? "#1e3a8a" : "#e2e8f0"}`,
          background: mode === "draw" ? "#eff6ff" : "#fff", color: mode === "draw" ? "#1e3a8a" : "#64748b",
        }}>✍️ Chora kwa Mkono</button>
        <button onClick={() => setMode("type")} style={{
          flex: 1, padding: "8px 12px", borderRadius: 8, cursor: "pointer", fontSize: 12, fontWeight: 700,
          border: `2px solid ${mode === "type" ? "#1e3a8a" : "#e2e8f0"}`,
          background: mode === "type" ? "#eff6ff" : "#fff", color: mode === "type" ? "#1e3a8a" : "#64748b",
        }}>⌨️ Andika Jina</button>
      </div>

      {mode === "draw" ? (
        <>
          <div style={{ fontSize: 11, color: "#64748b", marginBottom: 8 }}>
            Chora sahihi yako kwa kidole (simu) au mouse (kompyuta) hapa chini:
          </div>
          <div style={{ position: "relative", border: "2px dashed #cbd5e1", borderRadius: 12, background: "#fff", touchAction: "none" }}>
            <canvas
              ref={canvasRef}
              width={580}
              height={180}
              style={{ width: "100%", height: 140, cursor: "crosshair", display: "block", borderRadius: 10 }}
              onMouseDown={start}
              onMouseMove={draw}
              onMouseUp={stop}
              onMouseLeave={stop}
              onTouchStart={start}
              onTouchMove={draw}
              onTouchEnd={stop}
            />
            {!hasDrawn && (
              <div style={{
                position: "absolute", top: "50%", left: "50%", transform: "translate(-50%,-50%)",
                color: "#cbd5e1", fontSize: 13, pointerEvents: "none", fontStyle: "italic",
              }}>
                ✍️ Saini hapa…
              </div>
            )}
            <div style={{ position: "absolute", bottom: 8, left: 16, right: 16, borderBottom: "1px solid #e2e8f0" }} />
          </div>
          <div style={{ display: "flex", justifyContent: "flex-end", marginTop: 8 }}>
            <button onClick={clearCanvas} style={{
              background: "#fef2f2", border: "1px solid #fecaca", color: "#dc2626",
              borderRadius: 8, padding: "6px 14px", cursor: "pointer", fontSize: 12, fontWeight: 700,
            }}>🗑️ Futa Sahihi</button>
          </div>
        </>
      ) : (
        <>
          <div style={{ fontSize: 11, color: "#64748b", marginBottom: 8 }}>
            Andika jina lako, chagua mtindo unaopendelea:
          </div>
          <input
            type="text"
            value={typedName}
            placeholder="mfano: Amina H. Mwangi"
            onChange={e => { setTypedName(e.target.value); saveTyped(e.target.value, font); }}
            style={{
              width: "100%", boxSizing: "border-box", border: "2px solid #cbd5e1", borderRadius: 10,
              padding: "12px 14px", fontSize: 15, marginBottom: 10, outline: "none",
            }}
          />
          <div style={{ display: "flex", gap: 6, marginBottom: 12, flexWrap: "wrap" }}>
            {fonts.map(f => (
              <button key={f.id} onClick={() => { setFont(f.id); if (typedName) saveTyped(typedName, f.id); }} style={{
                border: `2px solid ${font === f.id ? "#1e3a8a" : "#e2e8f0"}`,
                background: font === f.id ? "#eff6ff" : "#fff",
                borderRadius: 8, padding: "6px 12px", cursor: "pointer", fontSize: 11, fontWeight: 600,
                color: "#374151",
              }}>{f.label}</button>
            ))}
          </div>
          {typedName && (
            <div style={{
              border: "2px dashed #cbd5e1", borderRadius: 12, padding: "20px 16px",
              background: "#fff", textAlign: "center", minHeight: 80, display: "flex", alignItems: "center", justifyContent: "center",
            }}>
              <span style={{ fontSize: 34, fontFamily: font, color: "#1e3a8a" }}>{typedName}</span>
            </div>
          )}
        </>
      )}
    </div>
  );
}


function downloadFile(title, content, signatureDataUrl) {
  const sigHtml = signatureDataUrl
    ? `<div style="margin-top:40px;">
         <img src="${signatureDataUrl}" style="height:70px;display:block;margin-bottom:2px;" />
         <div style="width:220px;border-top:1px solid #333;"></div>
       </div>`
    : "";
  const html = `<!DOCTYPE html>
<html lang="sw">
<head><meta charset="UTF-8"><title>${title}</title>
<style>
  body { font-family: 'Georgia', serif; margin: 70px; line-height: 1.9; color: #111; font-size: 12pt; max-width: 700px; }
  h1 { font-size: 17pt; color: #1e3a8a; border-bottom: 2px solid #1e3a8a; padding-bottom: 6px; }
  pre { white-space: pre-wrap; font-family: inherit; font-size: 11.5pt; }
  @media print { body { margin: 40px; } }
</style>
</head>
<body><h1>${title}</h1><pre>${content}</pre>${sigHtml}</body></html>`;
  const blob = new Blob([html], { type: "text/html" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = title.replace(/\s+/g, "_") + ".html";
  a.click();
  URL.revokeObjectURL(url);
}

            <div style={{ display: "flex", alignItems: "center", gap: 10, background: "#eff6ff", borderRadius: 10, padding: "10px 14px", marginTop: 16, border: "1px solid #bfdbfe", textAlign: "left" }}>
              <div style={{ width: 40, height: 40, borderRadius: "50%", background: "#dbeafe", display: "flex", alignItems: "center", justifyContent: "center", fontSize: 20, flexShrink: 0 }}>👩‍💼</div>
              <div>
                <div style={{ fontWeight: 700, color: "#1e40af", fontSize: 12 }}>Grace M. — Mtaalamu Mkuu, HirePro</div>
                <div style={{ color: "#3b82f6", fontSize: 11, marginTop: 2 }}>"Hati zako ziko tayari. Unaweza kuziedit hapa chini kisha upakue!"</div>
              </div>
            </div>
          </div>

          {/* ── SIGNATURE PAD ── */}
          <div style={{ ...card, marginBottom: 16 }}>
            <div style={{ fontWeight: 800, color: "#1e1b4b", fontSize: 15, marginBottom: 4, display: "flex", alignItems: "center", gap: 6 }}>
              ✍️ Weka Sahihi Yako
            </div>
            <p style={{ color: "#64748b", fontSize: 12, margin: "0 0 14px" }}>
              Ongeza sahihi yako halisi kwenye barua na/au CV — inaonyesha umakini na uhalisi.
            </p>

            <SignaturePad onSave={setSignature} savedSignature={signature} />

            {signature && (
              <div style={{ marginTop: 14 }}>
                <div style={{ fontSize: 12, fontWeight: 700, color: "#374151", marginBottom: 8 }}>Weka sahihi kwenye:</div>
                <div style={{ display: "flex", gap: 16 }}>
                  <label style={{ display: "flex", alignItems: "center", gap: 6, fontSize: 12, color: "#374151", cursor: "pointer" }}>
                    <input type="checkbox" checked={signOnLetter} onChange={e => setSignOnLetter(e.target.checked)} />
                    📨 Barua ya Maombi
                  </label>
                  <label style={{ display: "flex", alignItems: "center", gap: 6, fontSize: 12, color: "#374151", cursor: "pointer" }}>
                    <input type="checkbox" checked={signOnCV} onChange={e => setSignOnCV(e.target.checked)} />
                    📄 CV
                  </label>
                </div>

                <div style={{ marginTop: 12, background: "#f8fafc", border: "1px solid #e2e8f0", borderRadius: 10, padding: 14, textAlign: "center" }}>
                  <div style={{ fontSize: 11, color: "#94a3b8", marginBottom: 6 }}>Hivi ndivyo itakavyoonekana:</div>
                  <img src={signature} alt="Sahihi" style={{ height: 50, display: "inline-block" }} />
                  <div style={{ width: 160, borderTop: "1px solid #94a3b8", margin: "2px auto 0" }} />
                </div>
              </div>
            )}
          </div>

          {/* ── EDITOR: BARUA ── */}
          <div style={{ ...card, marginBottom: 16 }}>
            <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 12 }}>
              <div style={{ fontWeight: 800, color: "#1e1b4b", fontSize: 15, display: "flex", alignItems: "center", gap: 6 }}>
                📨 Barua ya Maombi
              </div>
              <div style={{ display: "flex", gap: 8 }}>
                <span style={{ fontSize: 10, background: "#dbeafe", color: "#1e40af", borderRadius: 20, padding: "2px 10px", fontWeight: 700 }}>
                  {coverLetter.length} herufi
                </span>
                <span style={{ fontSize: 10, background: "#dcfce7", color: "#15803d", borderRadius: 20, padding: "2px 10px", fontWeight: 700 }}>
                  ✏️ Inaweza kuhaririwa
                </span>
              </div>
            </div>

            {/* Toolbar */}
            <div style={{ display: "flex", gap: 6, marginBottom: 8, flexWrap: "wrap" }}>
              {[
                { label: "B", title: "Neno Zito", action: () => document.execCommand("bold") },
                { label: "I", title: "Italiki", action: () => document.execCommand("italic") },
                { label: "U̲", title: "Chini ya Mstari", action: () => document.execCommand("underline") },
              ].map(t => (
                <button key={t.label} title={t.title} onClick={t.action} style={{
                  background: "#f1f5f9", border: "1px solid #e2e8f0", borderRadius: 6,
                  width: 30, height: 28, cursor: "pointer", fontWeight: 800, fontSize: 13, color: "#374151"
                }}>{t.label}</button>
              ))}
              <div style={{ flex: 1 }} />
              <button onClick={() => setCoverLetter(coverLetter)} style={{
                background: "#fef9c3", border: "1px solid #fde68a", borderRadius: 6,
                padding: "4px 10px", cursor: "pointer", fontSize: 11, fontWeight: 700, color: "#92400e"
              }}>↺ Rejesha Asili</button>
            </div>

            <textarea
              value={coverLetter}
              onChange={e => setCoverLetter(e.target.value)}
              rows={14}
              style={{
                width: "100%", boxSizing: "border-box",
                border: "1px solid #bfdbfe", borderRadius: 10,
                padding: "14px 16px", fontSize: 12.5, lineHeight: 1.9,
                color: "#1a202c", fontFamily: "'Georgia', serif",
                background: "#fefefe", resize: "vertical", outline: "none",
              }}
            />
            <div style={{ display: "flex", gap: 8, marginTop: 10 }}>
              <button style={{ ...btnPrimary, marginTop: 0, flex: 1 }}
                onClick={() => downloadFile("Barua_ya_Maombi_ya_Kazi", coverLetter, signOnLetter ? signature : null)}>
                ⬇️ Pakua Barua
              </button>
            </div>
          </div>

          {/* ── EDITOR: CV ── */}
          <div style={{ ...card, marginBottom: 16 }}>
            <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 12 }}>
              <div style={{ fontWeight: 800, color: "#1e1b4b", fontSize: 15, display: "flex", alignItems: "center", gap: 6 }}>
                📄 CV Iliyosasishwa
              </div>
              <div style={{ display: "flex", gap: 8 }}>
                <span style={{ fontSize: 10, background: "#dbeafe", color: "#1e40af", borderRadius: 20, padding: "2px 10px", fontWeight: 700 }}>
                  {updatedCV.split("\n").length} mistari
                </span>
                <span style={{ fontSize: 10, background: "#dcfce7", color: "#15803d", borderRadius: 20, padding: "2px 10px", fontWeight: 700 }}>
                  ✏️ Inaweza kuhaririwa
                </span>
              </div>
            </div>

            {/* CV Section shortcuts */}
            <div style={{ display: "flex", gap: 6, marginBottom: 10, flexWrap: "wrap" }}>
              <div style={{ fontSize: 11, color: "#64748b", fontWeight: 600, width: "100%", marginBottom: 2 }}>Ruka sehemu:</div>
              {["MUHTASARI","UJUZI","UZOEFU","ELIMU"].map(s => (
                <button key={s} onClick={() => {
                  const ta = document.getElementById("cv-editor");
                  const idx = ta?.value.indexOf(s);
                  if (ta && idx >= 0) { ta.focus(); ta.setSelectionRange(idx, idx + s.length); ta.scrollTop = ta.scrollHeight * (idx / ta.value.length); }
                }} style={{
                  background: "#eff6ff", border: "1px solid #bfdbfe", borderRadius: 20,
                  padding: "3px 10px", cursor: "pointer", fontSize: 10, fontWeight: 700, color: "#1e40af"
                }}>{s}</button>
              ))}
              <div style={{ flex: 1 }} />
              <button onClick={() => {
                const original = updatedCV;
                setUpdatedCV(""); setTimeout(() => setUpdatedCV(original), 10);
              }} style={{
                background: "#fef9c3", border: "1px solid #fde68a", borderRadius: 6,
                padding: "4px 10px", cursor: "pointer", fontSize: 11, fontWeight: 700, color: "#92400e"
              }}>↺ Rejesha Asili</button>
            </div>

            {/* Tips */}
            <div style={{ background: "#f0fdf4", borderRadius: 8, padding: "8px 12px", marginBottom: 10, fontSize: 11, color: "#15803d", lineHeight: 1.6 }}>
              💡 <strong>Vidokezo vya kuhariri:</strong> Ongeza nambari kwenye mafanikio yako (mfano: "Nilisimamia timu ya watu 8" badala ya "Nilisimamia timu"). Hii inafanya CV yako kusimama zaidi.
            </div>

            <textarea
              id="cv-editor"
              value={updatedCV}
              onChange={e => setUpdatedCV(e.target.value)}
              rows={22}
              style={{
                width: "100%", boxSizing: "border-box",
                border: "1px solid #bfdbfe", borderRadius: 10,
                padding: "14px 16px", fontSize: 12.5, lineHeight: 1.9,
                color: "#1a202c", fontFamily: "'Courier New', monospace",
                background: "#fefefe", resize: "vertical", outline: "none",
              }}
            />

            {/* Word count and warnings */}
            <div style={{ display: "flex", justifyContent: "space-between", marginTop: 6, fontSize: 11, color: "#94a3b8" }}>
              <span>Maneno: {updatedCV.split(/\s+/).filter(Boolean).length}</span>
              <span style={{ color: updatedCV.length > 3000 ? "#ef4444" : "#94a3b8" }}>
                {updatedCV.length > 3000 ? "⚠️ CV ndefu sana — punguza hadi ukurasa 2" : "✓ Urefu mzuri"}
              </span>
            </div>

            <div style={{ display: "flex", gap: 8, marginTop: 10 }}>
              <button style={{ ...btnPrimary, marginTop: 0, flex: 1, background: "linear-gradient(135deg,#059669,#10b981)", boxShadow: "0 4px 14px rgba(5,150,105,0.35)" }}
                onClick={() => downloadFile("CV_Iliyosasishwa", updatedCV, signOnCV ? signature : null)}>
                ⬇️ Pakua CV
              </button>
            </div>
          </div>

          {/* Download both */}
          <div style={{ ...card, marginBottom: 16 }}>
            <button style={{ ...btnPrimary, marginTop: 0 }} onClick={() => {
              downloadFile("Barua_ya_Maombi_ya_Kazi", coverLetter, signOnLetter ? signature : null);
              setTimeout(() => downloadFile("CV_Iliyosasishwa", updatedCV, signOnCV ? signature : null), 800);
            }}>
              ⬇️ Pakua Hati Zote Mbili Pamoja
            </button>
            <button style={{ ...btnSecondary, marginTop: 10 }} onClick={reset}>
              🔄 Anza Upya — Maombi Mapya
            </button>
            <div style={{ textAlign: "center", marginTop: 14, color: "#94a3b8", fontSize: 11 }}>
              © 2026 HirePro Tanzania · Huduma za Kitaalamu za Maombi ya Kazi
            </div>
          </div>

        </div>
      )}
    </div>
  );
}
